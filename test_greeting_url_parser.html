// Extract greeting ID from QR code URL
const extractGreetingId = (data: string): string => {
  try {
    if (data.includes('http')) {
      const url = new URL(data);
      const id = url.searchParams.get('id');
      return id || data;
    }
    return data;
  } catch {
    return data;
  }
};
